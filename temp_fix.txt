          # Create docker-compose file
          cat > docker-compose.test.yml <<EOFCOMPOSE
          version: '3.8'
          services:
            db:
              image: mysql:8
              environment:
                MYSQL_ROOT_PASSWORD: test_password
                MYSQL_DATABASE: color_app_test
              healthcheck:
                test: ["CMD", "mysqladmin", "ping", "-h", "localhost"]
                interval: 10s
                timeout: 5s
                retries: 5

            app:
              image: $ECR_REGISTRY/$ECR_REPOSITORY:$IMAGE_TAG
              ports:
                - "80:3000"
              environment:
                DB_HOST: db
                DB_USER: root
                DB_PASSWORD: test_password
                DB_NAME: color_app_test
                SESSION_SECRET: test_secret
                NODE_ENV: production
              depends_on:
                db:
                  condition: service_healthy
          EOFCOMPOSE

          # Copy docker-compose file to EC2
          scp -i test-key.pem -o StrictHostKeyChecking=no docker-compose.test.yml ec2-user@$PUBLIC_IP:~/app/docker-compose.yml

          # Deploy on EC2
          ssh -i test-key.pem -o StrictHostKeyChecking=no ec2-user@$PUBLIC_IP <<'ENDSSH'
            cd ~/app
            # Login to ECR
            aws ecr get-login-password --region ${AWS_REGION} | sudo docker login --username AWS --password-stdin ${ECR_REGISTRY}
            sudo docker-compose pull
            sudo docker-compose up -d
            echo "⏳ Waiting for services to be healthy..."
            sleep 30
            sudo docker-compose ps
          ENDSSH

          echo "✅ Application deployed to temp EC2"
